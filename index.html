function preload() { this.load.image('sky', 'https://labs.phaser.io/assets/skies/space3.png'); // Добавляем проверку: если файл не найдется, Phaser выдаст ошибку в консоль // УБЕДИСЬ, ЧТО ИМЕНА НИЖЕ СОВПАДАЮТ С ИМЕНАМИ В GITHUB НА 100% this.load.image('pepe', 'pepe.png'); this.load.image('ton_coin', 'ton.png'); this.load.image('not_coin', 'not.png'); this.load.image('dogs_coin', 'dogs.png'); } function create() { this.add.image(config.width/2, config.height/2, 'sky').setAlpha(0.4).setScale(2); // ПЕПЕ: ставим фиксированный размер 100 пикселей plane = this.add.sprite(config.width / 2, 80, 'pepe'); plane.setDisplaySize(100, 100); plane.setDepth(5); claw = this.add.circle(config.width / 2, 140, 20, 0xffd700).setDepth(4); chain = this.add.graphics().setDepth(3); targets = this.physics.add.group(); const assets = ['ton_coin', 'dogs_coin', 'not_coin']; for(let i=0; i<15; i++) { let x = Phaser.Math.Between(40, config.width - 40); let y = Phaser.Math.Between(250, config.height - 100); let type = Phaser.Math.RND.pick(assets); let coin = targets.create(x, y, type); // Масштабируем монетки принудительно if (type === 'ton_coin') { coin.setDisplaySize(40, 40); // Маленький TON coin.setData('val', 50); } else { coin.setDisplaySize(55, 55); // Остальные coin.setData('val', 10); } } scoreText = this.add.text(20, 20, 'Score: 0', { fontSize: '24px', fill: '#fff', fontStyle: 'bold' }); timerText = this.add.text(config.width - 130, 20, 'Time: 60', { fontSize: '24px', fill: '#ff0', fontStyle: 'bold' }); this.time.addEvent({ delay: 1000, callback: () => { if (timeLeft > 0) timeLeft--; timerText.setText('Time: ' + timeLeft); }, loop: true }); this.input.on('pointerdown', () => { if(!isLaunching && !isReturning && timeLeft > 0) isLaunching = true; }); }
