function preload() { // Чтобы картинки не блокировались this.load.crossOrigin = 'anonymous'; // Загрузка (проверь, что в g..jpg реально две точки) this.load.image('sky', 'g..jpg'); this.load.image('pepe', 'pepe.png'); this.load.image('ton', 'ton-ecosystem-removebe.png'); // Если картинка не загрузится, создаем пустую текстуру, чтобы игра не упала this.load.on('loaderror', (file) => { console.warn('Не удалось загрузить: ' + file.key); }); } function create() { // 1. ФОН (с проверкой) if (this.textures.exists('sky')) { let bg = this.add.image(config.width/2, config.height/2, 'sky'); let scale = Math.max(config.width / bg.width, config.height / bg.height); bg.setScale(scale); } else { this.cameras.main.setBackgroundColor('#87CEEB'); // Голубое небо, если нет картинки } // 2. ПЕПЕ plane = this.add.sprite(config.width / 2, 120, 'pepe').setDepth(10); plane.setDisplaySize(200, 150); // Эффект полета Пепе (плавное движение вверх-вниз) this.tweens.add({ targets: plane, y: 130, duration: 2000, yoyo: true, loop: -1, ease: 'Sine.easeInOut' }); chain = this.add.graphics().setDepth(5); clawGraphic = this.add.graphics().setDepth(6); this.clawPoint = this.add.circle(config.width/2, 220, 15); this.physics.add.existing(this.clawPoint); // 3. МОНЕТЫ (улучшенная группа) targets = this.physics.add.group(); spawnCoins(); // Вынес в отдельную функцию для удобства // Тексты и Таймер (без изменений) scoreText = this.add.text(20, 30, 'Очки: 0', { fontSize: '24px', fill: '#fff', stroke: '#000', strokeThickness: 5 }); timerText = this.add.text(config.width - 160, 30, 'Время: 60', { fontSize: '24px', fill: '#ff0', stroke: '#000', strokeThickness: 5 }); this.time.addEvent({ delay: 1000, callback: () => { if(timeLeft > 0) { timeLeft--; timerText.setText('Время: ' + timeLeft); } }, loop: true }); this.input.on('pointerdown', () => { if(!isLaunching && !isReturning && timeLeft > 0) isLaunching = true; }); this.physics.add.overlap(this.clawPoint, targets, (c, target) => { if (isLaunching && !caughtItem) { caughtItem = target; isLaunching = false; isReturning = true; } }); } // Вспомогательная функция для создания монет function spawnCoins() { for(let i=0; i<8; i++) { let x = Phaser.Math.Between(50, window.innerWidth - 50); let y = Phaser.Math.Between(350, window.innerHeight - 100); let coin = targets.create(x, y, 'ton'); if(coin) coin.setDisplaySize(70, 70).setDepth(4); } }
